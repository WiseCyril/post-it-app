/**
 * @description Signout Route.
 * GET:/user/signout
 * @param {string} email; 
 *
 * @returns {Promise}
 */
 
export const signOut = (req, res) => {
  firebase.auth().signOut()
    .then(() => res.status(200).json({ message: 'signed-out successfully.' }))
    .catch(() => res.status(500).json({ message: 'Network Error' }));
};
